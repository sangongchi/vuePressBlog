### call

1. call() 改变 this 的指向

```js
//call 传的参数是一个个传递的

Function.prototype.myCall = function (context) {
  context = context || window;

  //首先使用this获取调用call的函数
  if (typeof this !== 'function') {
    throw new TypeError('Error');
  }
  context.fn = this;
  let args = [];
  // args = [...arguments].slice(1);
  args = Array.prototype.slice.call(arguments, 1);
  let result = context.fn(...args);
  delete context.fn;
  return result;
};

let obj = {
  name: '测试call',
};
function bar(a, b) {
  console.log(a, b);
  console.log(this.name);
}

bar.call(obj, 1, 2);
bar.myCall(obj, 123, 12);

```

### bind

```js
//call 传的参数是一个个传递的

Function.prototype.myBind = function (context) {
  context = context || window;

  //首先使用this获取调用call的函数
  if (typeof this !== 'function') {
    throw new TypeError('Error');
  }
  let _this = this;
  let args = [...arguments].slice(1);

  //bind 返回的是一个函数
  return function Fn() {
    //函数有两种调用方式，一种是new，一种是直接调用
    if (this instanceof Fn) {
      return new _this(...args, ...arguments);
    }
    return _this.apply(context, args.concat(...arguments));
  };
};

let obj = {
  name: '测试call',
};
function bar(a, b) {
  console.log(a);
  console.log(b);
  console.log(this.name);
}

bar.bind(obj, 1, 2)();
bar.myBind(obj, 1, 2)();
```

### 函数柯里化 curry

```js
function curring(fn) {
  var allArgs = [];
  return function next() {
    // var args = [].slice.call(arguments); //call的主要作用是调用数组的slice方法同时将arguments类数组转化为数组
    var args = [...arguments];
    if (args.length > 0) {
      allArgs = allArgs.concat(args);
      return next;
    } else {
      return fn.apply(null, allArgs);
    }
  };
}

var add = curring(function () {
  var sum = 0;
  for (var i = 0; i < arguments.length; i++) {
    sum += arguments[i];
  }
  return sum;
});

console.log(add(5)(6)(3)());
```

### 节流防抖

#### 1. 节流（Throttling）

![1594825729347](H:\biji\typora_imgs\1594825729347.png)

```js
function throttle(fn, wait) {
  let prev = new Date();
  return function () {
    let args = arguments;
    let now = new Date();
    if (now - prev > wait) {
      fn.apply(this, args);
      prev = new Date();
    }
  };
}
```



#### 2. 防抖 ( Debouncing )

- 当一次事件发生后，在一段时间内没有再次触发该事件（如有事件触发则重新计算时间），就处理最后一次发生的事件。

    ![1594823555934](H:\biji\typora_imgs\1594823555934.png)

    ```js
    function Debouncing(fn, wait = 50, immediate) {
      let timmer;
      return function () {
        if (immediate) {
          fn.apply(this, arguments);
        }
        if (timmer) {
          clearTimeout(timmer);
        }
        timmer = setTimeout(() => {
          fn.apply(this, arguments);
        }, wait);
      };
    }
    
    //一般用于防止 滚动条滚动，以及用户的点击造成事件执行过于频繁
    window.addEventListener('scroll', Debouncing(f1, 500));
    ```

- 将节流和防抖书写在一个函数中，通过变量切换

    ```js
    
    //将节流和防抖书写在一个函数内通过参数进行切换
    function thro_debounce(fn, wait, isDebounce) {
      let timer;
      let lastCall = 0;
      return function (...args) {
        if (isDebounce) {
          if (timmer) clearTimeout(timmer);
          timer = setTimeout(() => {
            fn.apply(this, ...args);
          }, wait);
        } else {
          let now = new Date().getTime();
          if (now - lastCall < wait) return;
          lastCall = now;
          fn(...args);
        }
      };
    }
    ```
